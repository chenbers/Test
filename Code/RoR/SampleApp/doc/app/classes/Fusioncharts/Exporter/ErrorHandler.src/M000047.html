    def self.meta2qs(meta,file_name_with_path)
        meta_values=""
        if(!file_name_with_path.nil?)
          meta_values+="&fileName="+file_name_with_path
        end
        meta_values+= "&width=" +meta ["width"]
        meta_values+= "&height=" +meta ["height"]
        meta_values+= "&DOMId=" +meta ["DOMId"]
        return meta_values
      end
